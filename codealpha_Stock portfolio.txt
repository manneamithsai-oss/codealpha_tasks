stocks = {
    "AAPL": 180,
    "TSLA": 250,
    "GOOG": 140,
    "MSFT": 330
}

total = 0

print("Available Stocks:", stocks)

n = int(input("How many different stocks do you own? "))

for i in range(n):
    name = input("Enter Stock Name: ").upper()
    qty = int(input("Enter Quantity: "))

    if name in stocks:
        total += stocks[name] * qty
    else:
        print("Stock not found!")

print("\nTotal Investment Value = $", total)

with open("portfolio.txt", "w") as file:
    file.write("Total Investment Value = $" + str(total))

print("Result saved in portfolio.txt")
